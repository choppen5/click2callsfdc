sforce.interaction.cti.enableClickToDial(); sforce.interaction.cti.onClickToDial(startCall); function startCall(response) { var result = JSON.parse(response.result); var cleanednumber = cleanFormatting(result.number); params = {"PhoneNumber": cleanednumber}; Twilio.Device.connect(params); } function cleanFormatting(number) { //changes a SFDC formatted US number, which would be 415-555-1212 return number.replace(' ','').replace('-','').replace('(','').replace(')','').replace('+',''); }
